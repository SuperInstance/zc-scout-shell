### 1. Build: Lightweight GitHub Trending Scraper (Python)

I'll create a simple script that fetches and parses GitHub's trending pages for a given language/date, focusing on repositories relevant to the fleet's domains (AI agents, ML, Rust, simulation).

**File:** `scout_tools/trending_scout.py`
```python
#!/usr/bin/env python3
"""
Scout Tool: GitHub Trending Scraper
Fetches trending repos for specified languages/domains.
Outputs structured data for fleet monitoring.
"""

import requests
from bs4 import BeautifulSoup
import json
from datetime import datetime
import time

def fetch_trending(language="", since="daily"):
    """
    Fetch trending repos from GitHub.
    Args:
        language: str, e.g., "python", "rust", "all"
        since: str, "daily", "weekly", "monthly"
    Returns:
        list of dicts with repo metadata
    """
    base_url = "https://github.com/trending"
    url = f"{base_url}/{language}?since={since}" if language else f"{base_url}?since={since}"
    
    headers = {
        "User-Agent": "Cocapn-Fleet-Scout/1.0 (Monitoring Agent)"
    }
    
    try:
        resp = requests.get(url, headers=headers, timeout=10)
        resp.raise_for_status()
    except Exception as e:
        print(f"Error fetching {url}: {e}")
        return []
    
    soup = BeautifulSoup(resp.text, 'html.parser')
    repos = []
    
    for article in soup.find_all('article', class_='Box-row'):
        # Extract repo name and owner
        h2 = article.find('h2', class_='h3')
        if not h2:
            continue
        a_tag = h2.find('a')
        if not a_tag:
            continue
        repo_name = a_tag.get('href', '').strip('/')
        
        # Description
        desc_elem = article.find('p', class_='col-9')
        description = desc_elem.text.strip() if desc_elem else ""
        
        # Language
        lang_elem = article.find('span', itemprop='programmingLanguage')
        language_name = lang_elem.text.strip() if lang_elem else "Unknown"
        
        # Stars and forks today
        stars_link = article.find('a', href=lambda x: x and '/stargazers' in x)
        stars_text = stars_link.text.strip() if stars_link else "0"
        # Clean stars text (e.g., "1,234 stars today")
        stars_today = stars_text.split()[0].replace(',', '')
        
        repos.append({
            "repo": repo_name,
            "description": description,
            "language": language_name,
            "stars_today": stars_today,
            "scraped_at": datetime.utcnow().isoformat() + "Z",
            "trending_period": since
        })
    
    return repos

def filter_relevant(repos, keywords=None):
    """Filter repos based on fleet-relevant keywords."""
    if keywords is None:
        keywords = ["agent", "ai", "llm", "autonomous", "simulation", "rust", "plato", "holodeck", "gpu", "torch", "training"]
    
    relevant = []
    for repo in repos:
        desc_lower = repo["description"].lower()
        repo_lower = repo["repo"].lower()
        lang_lower = repo["language"].lower()
        
        # Check if any keyword appears in description, repo name, or language
        if any(kw in desc_lower or kw in repo_lower or kw in lang_lower for kw in keywords):
            relevant.append(repo)
    
    return relevant

def main():
    print("Scout: Scanning GitHub Trending (daily)...")
    
    # Scan multiple languages of interest
    languages = ["", "python", "rust", "jupyter-notebook"]
    all_repos = []
    
    for lang in languages:
        print(f"  Fetching {lang if lang else 'all'}...")
        repos = fetch_trending(language=lang, since="daily")
        all_repos.extend(repos)
        time.sleep(1)  # Be polite
    
    # Deduplicate by repo name
    unique_repos = {r["repo"]: r for r in all_repos}.values()
    
    # Filter for fleet relevance
    relevant = filter_relevant(unique_repos)
    
    # Output
    timestamp = datetime.utcnow().strftime("%Y-%m-%d")
    output_file = f"scout_reports/trending_{timestamp}.json"
    
    # Ensure directory exists
    import os
    os.makedirs("scout_reports", exist_ok=True)
    
    with open(output_file, 'w') as f:
        json.dump({
            "metadata": {
                "scout_cycle": 12,
                "date": timestamp,
                "total_scraped": len(unique_repos),
                "relevant_count": len(relevant)
            },
            "relevant_repos": relevant
        }, f, indent=2)
    
    print(f"Scout report saved to {output_file}")
    print(f"Found {len(relevant)} relevant trending repos.")
    
    # Print top 5 for immediate awareness
    print("\nTop 5 relevant trending repos (by stars today):")
    sorted_repos = sorted(relevant, key=lambda x: int(x.get("stars_today", 0)), reverse=True)[:5]
    for i, repo in enumerate(sorted_repos, 1):
        print(f"  {i}. {repo['repo']} ({repo['language']}) - ★{repo['stars_today']} today")
        print(f"     {repo['description'][:100]}...")

if __name__ == "__main__":
    main()
```
